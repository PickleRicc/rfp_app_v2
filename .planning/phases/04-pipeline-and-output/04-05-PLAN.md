Key integration:
```typescript
import {
  PageTracker,
  condenseContent,
  convertToPdf
} from '@/lib/generation/pipeline';

// In the generation function:
const pageTracker = new PageTracker();

// Wire condense callback with correct signature
pageTracker.setCondenseCallback(
  async (content, targetPages, companyName) => {
    const result = await condenseContent(content, targetPages, companyName);
    return {
      condensedContent: result.condensedContent,
      success: result.confidence !== 'low'
    };
  },
  rfpAnalysis.companyName || 'Our Company'
);

// Set limits from RFP if available
if (rfpAnalysis.pageLimits) {
  pageTracker.setVolumeLimits(rfpAnalysis.pageLimits);
}

// After generating technical volume DOCX:
const technicalDocxBuffer = await generateTechnicalVolume(...);
pageTracker.addSection({
  sectionId: 'technical',
  volumeType: 'Technical',
  content: technicalContentText,
});

// IMMEDIATELY convert to PDF (locked decision: auto-generate during generation)
const technicalPdfResult = await convertToPdf(technicalDocxBuffer);
const technicalPdfBuffer = technicalPdfResult.success
  ? technicalPdfResult.pdfBuffer
  : null;

if (!technicalPdfResult.success) {
  console.warn(`PDF conversion failed for technical volume: ${technicalPdfResult.error}`);
}

// Store BOTH buffers in proposal
// Structure for storage:
interface VolumeOutput {
  docxBuffer: Buffer;
  pdfBuffer: Buffer | null;
  pdfError?: string;
}

// Repeat for each volume (Management, Past Performance, etc.)

// At end, get status:
const pageStatus = pageTracker.getStatus();
if (pageStatus.status === 'over') {
  console.warn(`Page limit exceeded: ${pageStatus.message}`);
}

// Store volumes with BOTH formats and page allocation in proposal:
await supabase.from('proposals').update({
  volumes: {
    technical: {
      docx: technicalDocxBuffer.toString('base64'),
      pdf: technicalPdfBuffer?.toString('base64') || null,
      pdfError: technicalPdfResult.success ? null : technicalPdfResult.error
    },
    // ... other volumes
  },
  page_allocation: pageStatus
}).eq('id', proposalId);
```

**CRITICAL:** PDFs are generated HERE during proposal generation, not later during download. The package-builder (Plan 04-04) will read the pre-generated PDFs from storage, not generate them on-demand.

Do NOT restructure the existing generator - add tracking and PDF conversion as parallel concerns.

import { useState } from 'react';
import { generateOutline, formatOutlineAsText, OutlineResult } from '@/lib/generation/pipeline';

// State for outline view
const [showOutline, setShowOutline] = useState(false);
const [outline, setOutline] = useState<OutlineResult | null>(null);

// Download button
<a

  {showOutline ? 'Hide Outline' : 'View Outline'}

{showOutline && proposal.outline && (
