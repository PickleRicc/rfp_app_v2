**CRITICAL: Condense Callback Wiring**

The setCondenseCallback method must accept a function matching the condenseContent signature:

```typescript
class PageTracker {
  private condenseCallback?: CondenseCallback;
  private companyName: string = '';

  setCondenseCallback(callback: CondenseCallback, companyName: string): void {
    this.condenseCallback = callback;
    this.companyName = companyName;
  }

  // When adding a section that would exceed limits:
  async addSection(allocation: PageAllocation): Promise<{
    added: boolean;
    condensed: boolean;
    status: PageLimitStatus
  }> {
    const estimatedPages = this.estimatePages(allocation.content);
    const newTotal = this.currentPages + estimatedPages;

    if (this.limitPages && newTotal > this.limitPages && this.condenseCallback) {
      // Calculate target pages (current limit minus already used, with 5% buffer)
      const targetPages = Math.floor((this.limitPages - this.currentPages) * 0.95);

      // Invoke condenseContent via callback
      const result = await this.condenseCallback(
        allocation.content,
        targetPages,
        this.companyName
      );

      if (result.success) {
        allocation.content = result.condensedContent;
        allocation.estimatedPages = this.estimatePages(result.condensedContent);
        // Add condensed content
        this.allocations.push(allocation);
        return { added: true, condensed: true, status: this.getStatus() };
      }
    }

    // Add as-is (even if over - let caller decide)
    this.allocations.push({ ...allocation, estimatedPages });
    return { added: true, condensed: false, status: this.getStatus() };
  }
}
```

This explicit wiring ensures PageTracker can invoke condenseContent with the correct parameters (content, targetPages, companyName).
